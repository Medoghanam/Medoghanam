( صحيح ) 🌺🌺🌺🌺🌺🌺🌺 ( سـبـحـان ﷲ و بـحـمـده ، عـدد خـلـقــه ، و رضا نفسه و زنة عرشه ، و مداد كلماته ) - ثلاث مرات ( صحيح ) 🌺🌺🌺🌺🌺🌺🌺 ( أصبحنا علي فطرة الإسلام ، و على كلمة إلاخلاص ، و على ديـــن نـبـيـنـــا مـحـمـــد ﷺ ، و على ملة أبينا إبراهيم حنيفاً مسلماً و مــا كــان مــن الـمـشــركـيــن ) - ( سنده صحيح ) 🌺🌺🌺🌺🌺🌺🌺 ( اللهمَّ إني أسألك العفو و العافية في الدنيا و الآخـرة ، اللهم إني أسألك العفو و العافية في دينـي و دنـيـاي ، و أهلي و مالي ، اللهم استر عوراتي ، و آمن روعاتي ، اللهم احفظني من بين يدي و من خلفي ، و عن يميني و عن شمالي ، و من فوقي ، و أعوذ بـعـظـمـتـك أن أغتال من تحتي ) - ( سنده صحيح ) 🌺🌺🌺🌺🌺🌺🌺 ( لا إله إلا ﷲ وحــده لا شــريــك له ، له الملك و له الحمد يحيي و يميت ، و هــو عـلـى كـــل شـــيء قــديـــر ) - عشر مرات ( حسن ) 🌺🌺🌺🌺🌺🌺🌺 ( بـســم ﷲ الــذي لا يـضــر مـع اسـمـــه شــيء في الأرض و لا في السماء و هو السميع العليم ) - ثلاث مرات ( حسن لغيره ) 🌺🌺🌺🌺🌺🌺🌺 دعاء سيد الإستغفار ( اللهم أنت ربي لا إله إلا أنت خلقتني و أنـا عـبـدك ، و أنـا عـلـى عــهـــدك و وعـــدك مــا اسـتـطـعـــت ، أعوذ بك من شر ما صنعت ، أبوء لك بنعمتك علي ، و أبوء بذنبي فاغفر لي فإنه لا يغفر الذنوب إلا أنت ) - ( صحيح البخاري ) 🌺🌺🌺🌺🌺🌺🌺 ( سـبـحـان ﷲ و بـحـمـده ) - مائة مرة 🌺🌺🌺🌺🌺🌺🌺 ( أعوذ بكلمات ﷲ التامات من شر ما خلق ) - ثلاث مرات 🌺🌺🌺🌺🌺🌺🌺 (( أَصـبَـحـنَـا وَ أَصـبَـحَ الـمُـــلكُ لِلَّهِ ، وَ الحَـمـدُ لِلَّهِ لَا إِلَهَ إِلَّا اللهُ ، وَحدَهُ لَا شَرِيكَ لَهُ ، لَهُ المُلكُ وَ لَهُ الحَمدُ وَ هُـوَ عَلَى كُلِّ شَيءٍ قَدِيرٌ ، رَبِّ أَســأَلُـكَ خَـيـرَ مَا فِي هَذِا اليوم وَ خَيرَ مَا بَعدَهَ ، وَ أَعُوذُ بِكَ مِن شَرِّ مَا فِي هَذِا اليوم وَ شَــرِّ مَا بَعدَهَ ، رَبِّ أَعُوذُ بِكَ مِنَ الكَسَلِ وَ سُوءِ الكِبَرِ ، رَبِّ أَعُوذُ بِكَ مِن عَذَابٍ فِي النَّارِ وَ عَذَابٍ فِي القَبرِ )) ، وَ إِذَا أَمسَى قَالَ ذَلِكَ أَيضاً : (( أَمـسَـيـنَـــا وَ أَمـسَــى الـمُـــلكُ لِلَّهِ ، وَ الحَمدُ لِلَّهِ لَا إِلَهَ إِلَّا اللهُ ، وَحدَهُ لَا شَرِيكَ لَهُ ، لَهُ المُلكُ وَ لَهُ الحَمدُ وَ هُوَ عَلَى كُلِّ شَيءٍ قَدِيرٌ ، رَبِّ أَســأَلُـكَ خَـيـرَ مَا فِي هَذِهِ اللَّيلَةِ وَ خَيرَ مَا بَعدَهَا ، وَ أَعُـــوذُ بِكَ مِن شَرِّ مَا فِي هَذِهِ اللَّيلَةِ وَ شَرِّ مَا بَعدَهَا ، رَبِّ أَعُوذُ بِكَ مِنَ الكَسَلِ وَ سُوءِ الكِبَرِ ، رَبِّ أَعُوذُ بِكَ مِن عَذَابٍ فِي النَّارِ وَ عَذَابٍ فِي القَبرِ )) ، -
